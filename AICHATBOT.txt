        <button onclick="sendMessage()">Send</button>
    </div>

<script>
function sendMessage() {
    const input = document.getElementById("user-input");
    const message = input.value.trim();
    const chatBox = document.getElementById("chat-box");

    if (message === "") return;

    chatBox.innerHTML += `<div class="user"><b>You:</b> ${message}</div>`;

    let reply = generateReply(message);

    setTimeout(() => {
        chatBox.innerHTML += `<div class="bot"><b>Bot:</b> ${reply}</div>`;
        chatBox.scrollTop = chatBox.scrollHeight;
    }, 500);

    input.value = "";
}

function handleEnter(event) {
    if (event.key === "Enter") {
        sendMessage();
    }
}

function generateReply(message) {
    message = message.toLowerCase();

    if (message.includes("hello") || message.includes("hi")) {
        return "Hello! How can I help you?";
    }
    else if (message.includes("project")) {
        return "Are you asking about your Final Year Project?";
    }
    else if (message.includes("name")) {
        return "I am your AI Chatbot.";
    }
    else if (message.includes("bye")) {
        return "Goodbye! Have a nice day.";
    }
    else {
        return "Sorry, I don't understand. Please ask something else.";
    }
}
